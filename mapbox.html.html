  <script src="https://api.mapbox.com/mapbox-gl-js/v2.14.1/mapbox-gl.js"></script>

  <link href="https://api.mapbox.com/mapbox-gl-js/v2.14.1/mapbox-gl.css" rel="stylesheet" />

  <style>

    body { margin: 0; padding: 0; }

    #map { position: absolute; top: 0; bottom: 0; width: 100%; }

  </style>

</head>

<body>

  <div id="map"></div>

  <script>

    mapboxgl.accessToken = 'pk.eyJ1IjoiYXNodG9uLW1vcnJvdyIsImEiOiJjbTllNGxrZG8xOXRxMmpvdm52ZWpidmw3In0.nmN1vCpni_NNgZDqChZQ0w';

    const map = new mapboxgl.Map({

      container: 'map',

      style: 'mapbox://styles/mapbox/light-v11',

      zoom: 1.5,

      center: [0, 20]

    });

    map.on('load', () => {

      map.addSource('countries', {

        type: 'vector',

        url: 'mapbox://mapbox.country-boundaries-v1'

      });

      map.addLayer({

        id: 'country-fills',

        type: 'fill',

        source: 'countries',

        'source-layer': 'country_boundaries',

        paint: {

          'fill-color': '#ccc',

          'fill-opacity': 0.5

        }

      });

      map.addLayer({

        id: 'country-fills-hover',

        type: 'fill',

        source: 'countries',

        'source-layer': 'country_boundaries',

        paint: {

          'fill-color': '#007aff',

          'fill-opacity': 0.7

        },

        filter: ['==', 'iso_3166_1_alpha_3', '']

      });

      let hoveredCountryId = null;

      map.on('mousemove', 'country-fills', (e) => {

        if (e.features.length > 0) {

          const countryId = e.features[0].properties.iso_3166_1_alpha_3;

          map.setFilter('country-fills-hover', ['==', 'iso_3166_1_alpha_3', countryId]);

          hoveredCountryId = countryId;

        }

      });

      map.on('mouseleave', 'country-fills', () => {

        map.setFilter('country-fills-hover', ['==', 'iso_3166_1_alpha_3', '']);

        hoveredCountryId = null;

      });

      map.on('click', 'country-fills', (e) => {

        const name = e.features[0].properties.name_en.toLowerCase().replace(/\s+/g, '-');

        window.location.href = `https://domain.com/${name}`;

      });

      map.getCanvas().style.cursor = 'default';

    });

  </script>

</body>
